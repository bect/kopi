{{- /* Image Render Hook This template overrides the default Markdown image rendering. It creates a responsive element with WebP sources and a fallback JPEG/PNG. It also adds lazy loading and proper dimensions to prevent layout shift. Usage in Markdown: ![Alt Text](/path/to/image.jpg "Optional Title") */ -}} {{- $src := .Destination -}} {{- $alt := .Text | plainify -}} {{- $title := .Title | plainify -}} {{- if or (strings.HasPrefix $src "http://") (strings.HasPrefix $src "https://") -}} {{- /* Handle external images */ -}} {{- else -}} {{- /* Handle local images (Page or Global Resources) */ -}} {{- $image := "" -}} {{- if strings.HasPrefix $src "/" -}} {{- $image = resources.Get (strings.TrimPrefix "/" $src) -}} {{- else -}} {{- $image = .Page.Resources.GetMatch $src -}} {{- end -}} {{- if not $image -}} {{- /* Fallback for local images not found in resources (e.g., in /static) */ -}} {{- else -}} {{- /* Process local resources to create responsive element */ -}} {{- $widths := slice 480 800 1200 -}} {{- /* Generate WebP sources */ -}} {{- $webp_sources := slice -}} {{- range $widths -}} {{- $resized := $image.Resize (printf "%dx" .) -}} {{- $webp := $resized.Process "webp" -}} {{- $webp_sources = $webp_sources | append (printf "%s %dw" $webp.Permalink .) -}} {{- end -}} {{- /* Generate original format sources (e.g., JPEG, PNG) */ -}} {{- $orig_sources := slice -}} {{- range $widths -}} {{- $resized := $image.Resize (printf "%dx" .) -}} {{- $orig_sources = $orig_sources | append (printf "%s %dw" $resized.Permalink .) -}} {{- end -}} {{- end -}} {{- end -}}
